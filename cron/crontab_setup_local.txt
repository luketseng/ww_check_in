# WW auto check-in (LOCAL) - crontab examples
# Format: minute hour day month weekday command

# Set your project directory (adjust path) and script
# Example environment overrides: PROJECT_DIR, RUN_SCRIPT, HEADLESS

# Example: One-time check-in (Aug 7 at 08:30)
# 30 8 7 8 * PROJECT_DIR=/home/you/ww_check_in HEADLESS=true /home/you/ww_check_in/cron/cron_check_in_local.sh check-in

# Example: Mon-Thu check-in at 08:30 (explicit check-in)
# 30 8 * * 1-4 PROJECT_DIR=/home/you/ww_check_in HEADLESS=true /home/you/ww_check_in/cron/cron_check_in_local.sh check-in

# Example: Mon-Thu check-out at 22:00
# 0 22 * * 1-4 PROJECT_DIR=/home/you/ww_check_in HEADLESS=true /home/you/ww_check_in/cron/cron_check_in_local.sh check-out

# Prefer .env over environment (recommended):
# 30 8 * * 1-4 PROJECT_DIR=/home/you/ww_check_in PREFER_DOTENV=true /home/you/ww_check_in/cron/cron_check_in_local.sh check-in
30 8 * * 1-4 $HOME/git/ww_check_in/cron/cron_check_in_local.sh
20 19 * * 1,4 $HOME/git/ww_check_in/cron/cron_check_in_local.sh
0 2 * * * find $HOME/git/ww_check_in/logs -name "*.log" -type f -mtime +5 -delete

# Use custom entry script (rare):
# 30 8 * * 1-4 PROJECT_DIR=/home/you/ww_check_in RUN_SCRIPT=test_check_in_complete_v2.py /home/you/ww_check_in/cron/cron_check_in_local.sh
